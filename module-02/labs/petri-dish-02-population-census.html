What You're Measuring

Building a list of N items takes N operations (one per item). That's O(N) -- linear time. Double N, double the work.

Linear search checks each item one by one. Worst case: check all N items. Average case: N/2. Best case: 1 (found immediately). This is also O(N).

Watch the comparison counter increment as each critter is checked. The orange flashes show each comparison. A magenta flash means "found it!" A red flash means "not found."

The graph plots your experimental data. If it looks like a straight line, you've confirmed linear time complexity.

""" critter_census.py - CSC 249 Petri Dish 02 List construction and linear search, measured. Run: python critter_census.py """ import random import time class CritterData: """Data-only version of the visual Critter.""" def __init__(self, id, value): self.id = id self.value = value def __str__(self): return f"Critter #{self.id} (value={self.value})" def build_population(n): """Build a list of n CritterData objects. Returns (list, operation_count).""" population = [] ops = 0 for i in range(n): value = random.randint(0, 100) population.append(CritterData(i, value)) ops += 1 # Each append is one operation return population, ops def linear_search(population, target_value): """Search for target_value in population. Returns (index or -1, comparison_count).""" comparisons = 0 for i, critter in enumerate(population): comparisons += 1 if critter.value == target_value: return i, comparisons return -1, comparisons # === EXPERIMENT RUNNER === if __name__ == "__main__": random.seed(42) print("=" * 50) print("LIST CONSTRUCTION: Operations vs N") print("=" * 50) print(f"{'N':>6} {'Ops':>8} {'Time(ms)':>10}") print("-" * 30) for n in [10, 20, 30, 40, 50, 100, 500, 1000]: start = time.time() pop, ops = build_population(n) elapsed = (time.time() - start) * 1000 print(f"{n:>6} {ops:>8} {elapsed:>10.3f}") print() print("=" * 50) print("LINEAR SEARCH: Comparisons vs N") print("=" * 50) print(f"{'N':>6} {'Best':>6} {'Worst':>6} {'Avg(10 trials)':>15}") print("-" * 40) for n in [10, 20, 50, 100, 500, 1000]: pop, _ = build_population(n) # Best case: search for the first element's value _, best = linear_search(pop, pop[0].value) # Worst case: search for a value that doesn't exist _, worst = linear_search(pop, -1) # Average case: search for 10 random existing values total_comps = 0 for _ in range(10): target = random.choice(pop).value _, comps = linear_search(pop, target) total_comps += comps avg = total_comps / 10 print(f"{n:>6} {best:>6} {worst:>6} {avg:>15.1f}") # TODO: Add your own experiments here # Try: What happens to average comparisons as N grows? # Try: Plot the results (optional, use matplotlib if available)

// critter_census.cpp - CSC 249 Petri Dish 02 // List construction and linear search, measured. // Compile: g++ -o critter_census critter_census.cpp // Run: ./critter_census #include <iostream> #include <vector> #include <string> #include <cstdlib> #include <ctime> #include <chrono> #include <iomanip> struct CritterData { int id; int value; std::string toString() const { return "Critter #" + std::to_string(id) + " (value=" + std::to_string(value) + ")"; } }; // Build a list of n CritterData. Returns operation count. int buildPopulation(std::vector<CritterData>& population, int n) { population.clear(); int ops = 0; for (int i = 0; i < n; i++) { CritterData c; c.id = i; c.value = rand() % 101; population.push_back(c); ops++; // Each push_back is one operation } return ops; } // Linear search. Returns comparison count. Sets foundIndex to index or -1. int linearSearch(const std::vector<CritterData>& pop, int targetValue, int& foundIndex) { int comparisons = 0; foundIndex = -1; for (int i = 0; i < (int)pop.size(); i++) { comparisons++; if (pop[i].value == targetValue) { foundIndex = i; return comparisons; } } return comparisons; } int main() { srand(time(nullptr)); std::cout << std::string(50, '=') << std::endl; std::cout << "LIST CONSTRUCTION: Operations vs N" << std::endl; std::cout << std::string(50, '=') << std::endl; std::cout << std::setw(6) << "N" << std::setw(8) << "Ops" << std::setw(12) << "Time(ms)" << std::endl; std::cout << std::string(30, '-') << std::endl; int sizes[] = {10, 20, 30, 40, 50, 100, 500, 1000}; for (int n : sizes) { std::vector<CritterData> pop; auto start = std::chrono::high_resolution_clock::now(); int ops = buildPopulation(pop, n); auto end = std::chrono::high_resolution_clock::now(); double elapsed = std::chrono::duration<double, std::milli>(end - start).count(); std::cout << std::setw(6) << n << std::setw(8) << ops << std::setw(12) << std::fixed << std::setprecision(3) << elapsed << std::endl; } std::cout << std::endl; std::cout << std::string(50, '=') << std::endl; std::cout << "LINEAR SEARCH: Comparisons vs N" << std::endl; std::cout << std::string(50, '=') << std::endl; std::cout << std::setw(6) << "N" << std::setw(8) << "Best" << std::setw(8) << "Worst" << std::setw(16) << "Avg(10)" << std::endl; std::cout << std::string(40, '-') << std::endl; for (int n : sizes) { std::vector<CritterData> pop; buildPopulation(pop, n); int idx; // Best case: first element int best = linearSearch(pop, pop[0].value, idx); // Worst case: value not in list int worst = linearSearch(pop, -1, idx); // Average: 10 random searches int totalComps = 0; for (int t = 0; t < 10; t++) { int target = pop[rand() % n].value; totalComps += linearSearch(pop, target, idx); } double avg = totalComps / 10.0; std::cout << std::setw(6) << n << std::setw(8) << best << std::setw(8) << worst << std::setw(16) << std::fixed << std::setprecision(1) << avg << std::endl; } // TODO: Add your own experiments // Try: What happens to average comparisons as N grows? return 0;
