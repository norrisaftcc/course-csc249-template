What You're Seeing

Selection Sort scans the unsorted region for the minimum value, then swaps it into position. Each scan is the inner loop (orange flashes). The swap is the magenta flash. This takes N*(N-1)/2 comparisons: O(N²).

After sorting, the critters line up left-to-right by value in a neat row at the bottom.

Binary Search only works on sorted data. It checks the middle, then discards half the remaining items. Each step halves the search space (dimmed critters are eliminated). This takes ~log&sub2;(N) comparisons: O(log N).

For N=50: Linear search needs up to 50 comparisons. Binary search needs at most 6. That's the power of sorting + binary search.

""" critter_classification.py - CSC 249 Petri Dish 03 Selection sort and binary search, measured. Run: python critter_classification.py """ import random import time class CritterData: def __init__(self, id, value): self.id = id self.value = value def __str__(self): return f"Critter #{self.id} (value={self.value})" def selection_sort(population): """Sort population by value using selection sort. Returns (comparisons, swaps).""" comparisons = 0 swaps = 0 n = len(population) for i in range(n - 1): min_idx = i for j in range(i + 1, n): comparisons += 1 if population[j].value < population[min_idx].value: min_idx = j if min_idx != i: population[i], population[min_idx] = population[min_idx], population[i] swaps += 1 return comparisons, swaps def linear_search(population, target_value): """Linear search. Returns (index or -1, comparisons).""" comparisons = 0 for i, critter in enumerate(population): comparisons += 1 if critter.value == target_value: return i, comparisons return -1, comparisons def binary_search(sorted_pop, target_value): """Binary search on sorted population. Returns (index or -1, comparisons).""" comparisons = 0 low, high = 0, len(sorted_pop) - 1 while low <= high: mid = (low + high) // 2 comparisons += 1 if sorted_pop[mid].value == target_value: return mid, comparisons elif sorted_pop[mid].value < target_value: low = mid + 1 else: high = mid - 1 return -1, comparisons # === EXPERIMENTS === if __name__ == "__main__": random.seed(42) print("=" * 55) print("SELECTION SORT: Comparisons vs N") print("=" * 55) print(f"{'N':>6} {'Comps':>8} {'Swaps':>6} {'N*(N-1)/2':>10} {'Time(ms)':>10}") print("-" * 48) for n in [10, 20, 30, 40, 50]: pop = [CritterData(i, random.randint(0, 100)) for i in range(n)] start = time.time() comps, swaps = selection_sort(pop) elapsed = (time.time() - start) * 1000 expected = n * (n - 1) // 2 print(f"{n:>6} {comps:>8} {swaps:>6} {expected:>10} {elapsed:>10.3f}") print() print("=" * 55) print("LINEAR vs BINARY SEARCH (on sorted data)") print("=" * 55) print(f"{'N':>6} {'Linear(worst)':>14} {'Binary(worst)':>14} {'log2(N)':>8}") print("-" * 48) import math for n in [10, 20, 50, 100, 500, 1000]: pop = [CritterData(i, i) for i in range(n)] # sorted 0..n-1 _, lin_comps = linear_search(pop, -1) # worst case _, bin_comps = binary_search(pop, -1) # worst case log_n = math.ceil(math.log2(n)) if n > 0 else 0 print(f"{n:>6} {lin_comps:>14} {bin_comps:>14} {log_n:>8}") # TODO: Try insertion sort and compare to selection sort # TODO: Test binary search on nearly-sorted vs fully-sorted data
APPENDIX B: C++ - Selection Sort & Binary Search
// critter_classification.cpp - CSC 249 Petri Dish 03 // Selection sort and binary search, measured. // Compile: g++ -o critter_class critter_classification.cpp // Run: ./critter_class #include <iostream> #include <vector> #include <cstdlib> #include <ctime> #include <chrono> #include <iomanip> #include <cmath> struct CritterData { int id; int value; }; // Selection sort. Returns {comparisons, swaps}. std::pair<int,int> selectionSort(std::vector<CritterData>& pop) { int comps = 0, swaps = 0; int n = pop.size(); for (int i = 0; i < n - 1; i++) { int minIdx = i; for (int j = i + 1; j < n; j++) { comps++; if (pop[j].value < pop[minIdx].value) { minIdx = j; } } if (minIdx != i) { std::swap(pop[i], pop[minIdx]); swaps++; } } return {comps, swaps}; } // Linear search. Returns {index (-1 if not found), comparisons}. std::pair<int,int> linearSearch(const std::vector<CritterData>& pop, int target) { int comps = 0; for (int i = 0; i < (int)pop.size(); i++) { comps++; if (pop[i].value == target) return {i, comps}; } return {-1, comps}; } // Binary search. Returns {index (-1 if not found), comparisons}. std::pair<int,int> binarySearch(const std::vector<CritterData>& pop, int target) { int comps = 0; int low = 0, high = (int)pop.size() - 1; while (low <= high) { int mid = (low + high) / 2; comps++; if (pop[mid].value == target) return {mid, comps}; else if (pop[mid].value < target) low = mid + 1; else high = mid - 1; } return {-1, comps}; } int main() { srand(time(nullptr)); std::cout << std::string(55, '=') << std::endl; std::cout << "SELECTION SORT: Comparisons vs N" << std::endl; std::cout << std::string(55, '=') << std::endl; std::cout << std::setw(6) << "N" << std::setw(10) << "Comps" << std::setw(8) << "Swaps" << std::setw(12) << "N*(N-1)/2" << std::endl; std::cout << std::string(40, '-') << std::endl; int sizes[] = {10, 20, 30, 40, 50}; for (int n : sizes) { std::vector<CritterData> pop; for (int i = 0; i < n; i++) pop.push_back({i, rand() % 101}); auto [comps, swaps] = selectionSort(pop); int expected = n * (n - 1) / 2; std::cout << std::setw(6) << n << std::setw(10) << comps << std::setw(8) << swaps << std::setw(12) << expected << std::endl; } std::cout << std::endl; std::cout << std::string(55, '=') << std::endl; std::cout << "LINEAR vs BINARY SEARCH" << std::endl; std::cout << std::string(55, '=') << std::endl; std::cout << std::setw(6) << "N" << std::setw(16) << "Linear(worst)" << std::setw(16) << "Binary(worst)" << std::setw(10) << "log2(N)" << std::endl; std::cout << std::string(48, '-') << std::endl; int searchSizes[] = {10, 20, 50, 100, 500, 1000}; for (int n : searchSizes) { std::vector<CritterData> pop; for (int i = 0; i < n; i++) pop.push_back({i, i}); auto [li, lc] = linearSearch(pop, -1); auto [bi, bc] = binarySearch(pop, -1); int logN = (int)ceil(log2(n)); std::cout << std::setw(6) << n << std::setw(16) << lc << std::setw(16) << bc << std::setw(10) << logN << std::endl; } // TODO: Implement insertion sort and compare return 0; }
