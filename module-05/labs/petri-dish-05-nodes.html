What You're Seeing

Linked List [val | next]: Each box shows a node split into two fields: the value and the next pointer. Nodes live anywhere in memory — the pointer is the only connection. Insert Head rewires one pointer: O(1). Insert Tail must walk the entire chain first: O(n). Search checks nodes one by one with no shortcut: always O(n). Click Step ▶ repeatedly to walk through each node; watch the cursor advance through the chain.

BST [left | val | right]: The same node idea, but each node carries two pointers. That extra pointer enables branching: at each node, search discards half the remaining tree. Inorder ▶ steps through nodes in sorted order (left → root → right), visiting every node exactly once. Try Insert 1–10 (sorted): the BST degenerates into a right-leaning chain — effectively a linked list — and search cost rises to O(n).

// critter_nodes.cpp - CSC 249 Petri Dish 05 // Linked List and Binary Search Tree implementations. // Compile: g++ -std=c++17 -o critter_nodes critter_nodes.cpp // Run: ./critter_nodes // // NODE ANATOMY // Linked List Node: Binary Tree Node: // +--------+------+ +------+--------+-------+ // | value | next | | left | value | right | // +--------+------+ +------+--------+-------+ // (1 pointer) (2 pointers) #include <iostream> #include <vector> #include <string> #include <cstdlib> #include <ctime> struct CritterData { int id, value; std::string toString() const { return "#" + std::to_string(id) + "(v=" + std::to_string(value) + ")"; } }; // ============================================================ // LINKED LIST // ============================================================ struct ListNode { CritterData critter; ListNode* next; // ONE pointer: the only connection ListNode(CritterData c) : critter(c), next(nullptr) {} }; class LinkedList { ListNode* head = nullptr; int _size = 0; void destroy(ListNode* n) { if (!n) return; destroy(n->next); delete n; } public: ~LinkedList() { destroy(head); } // O(1) -- one pointer update void insertHead(CritterData c) { ListNode* n = new ListNode(c); n->next = head; head = n; _size++; } // O(n) -- walk to tail first void insertTail(CritterData c) { ListNode* n = new ListNode(c); if (!head) { head = n; _size++; return; } ListNode* cur = head; while (cur->next) cur = cur->next; // walk, walk, walk... cur->next = n; _size++; } // O(1) -- one pointer update, then free CritterData removeHead() { if (!head) throw std::runtime_error("removeHead on empty list"); ListNode* old = head; CritterData removed = old->critter; head = head->next; delete old; // C++ requires manual deallocation _size--; return removed; } // O(n) -- no shortcut; every node is a candidate std::pair<bool, int> search(int targetValue) { int comps = 0; ListNode* cur = head; while (cur) { comps++; if (cur->critter.value == targetValue) return {true, comps}; cur = cur->next; } return {false, comps}; } std::vector<CritterData> traverse() const { std::vector<CritterData> result; ListNode* cur = head; while (cur) { result.push_back(cur->critter); cur = cur->next; } return result; } int size() const { return _size; } }; // ============================================================ // BINARY SEARCH TREE // ============================================================ struct TreeNode { CritterData critter; TreeNode* left = nullptr; // TWO pointers: branching TreeNode* right = nullptr; // makes O(log n) possible TreeNode(CritterData c) : critter(c) {} }; class BST { TreeNode* root = nullptr; int _size = 0; void inorderHelper(TreeNode* n, std::vector<CritterData>& r) const { if (!n) return; inorderHelper(n->left, r); // left subtree first r.push_back(n->critter); inorderHelper(n->right, r); // right subtree last } void destroy(TreeNode* n) { if (!n) return; destroy(n->left); destroy(n->right); delete n; } public: ~BST() { destroy(root); } int insert(CritterData c) { int comps = 0; if (!root) { root = new TreeNode(c); _size++; return comps; } TreeNode* cur = root; while (true) { comps++; if (c.value < cur->critter.value) { if (!cur->left) { cur->left = new TreeNode(c); _size++; return comps; } cur = cur->left; } else { if (!cur->right) { cur->right = new TreeNode(c); _size++; return comps; } cur = cur->right; } } } std::pair<bool, int> search(int target) const { int comps = 0; TreeNode* cur = root; while (cur) { comps++; if (target == cur->critter.value) return {true, comps}; cur = target < cur->critter.value ? cur->left : cur->right; } return {false, comps}; } std::vector<CritterData> inorder() const { std::vector<CritterData> result; inorderHelper(root, result); return result; } int size() const { return _size; } }; // ============================================================ // DEMO: same data, two structures, different behavior // ============================================================ int main() { srand(time(nullptr));
